def clock_exist?
  unless defined?(@clock_exist_p)
    @clock_exist_p = @move_infos.any? { |e| e[:used_seconds].to_i.nonzero? }
  end

  @clock_exist_p
end

def error_message_part(comment_mark = "*")
  if v = error_message
    v = v.strip + "\n"
    s = "-" * 76 + "\n"
    [s, *v.lines, s].collect { |e| "#{comment_mark} #{e}" }.join
  end
end
