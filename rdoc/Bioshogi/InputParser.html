def check(text)
  rows = InputParser.scan(text).collect do |str|
    { "入力" => str }.merge(match!(str).named_captures)
  end
  tp rows
end

def match!(str)
  md = str.to_s.match(regexp)
  unless md
    raise SyntaxDefact, "表記が間違っています : #{str.inspect}"
  end
  md
end
