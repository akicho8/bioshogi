      cold_war_verification (e)
      Source 
      # File lib/bioshogi/analysis/executer_dsl.rb, line 60
def cold_war_verification(e)
  # 開戦済みならskip
  # 例えばそれまで静かで始めて角交換したときは kill_count は 1 になるので outbreak_skip: nil, kill_count_lteq: 1 にしておけば skip されない
  # 0 < 1キル → skip
  # 1 < 1キル → ok
  if v = e.kill_count_lteq
    if v >= container.kill_count
    else
      throw :skip
    end
  end

  # 「歩と角」を除く駒が取られた場合はskip
  if e.outbreak_skip
    if container.outbreak_turn # 「歩と角」を除く駒が取られた手数が入っている
      throw :skip
    end
  end

  # 歩を除いて何か持っていたらskip
  if e.has_other_pawn_then_skip
    if !piece_box.except(:pawn).empty?
      throw :skip
    end
  end
end

def left_or_light_from_a_to_b(a, b)
  v1 = a.white_then_flip.place.column.value
  v2 = b.white_then_flip.place.column.value
  case
  when v1 < v2
    :left
  when v1 > v2
    :right
  end
end

def tag_add(tag, options = {})
  tag = TagIndex.fetch(tag)

  if options[:once] && player.tag_bundle.include?(tag)
    return
  end

  player.tag_bundle << tag
  tag_bundle << tag

  if v = tag.add_to_self
    player.tag_bundle << v
  end

  if v = tag.add_to_opponent
    opponent_player.tag_bundle << v
  end
end
