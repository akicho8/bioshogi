def fuzzy_lookup(key)
  found = nil
  AbbrevExpander.expand(key).each do |str|
    if found = lookup(str)
      break
    end
  end
  found
end

def values_hash
  @values_hash ||= values.inject({}) do |a, e|
    [e.key, *e.alias_names].inject(a) do |a, key|
      key = key.to_sym
      if a.has_key?(key)
        raise "must not happen"
      end
      a.merge(key => e)
    end
  end
end
