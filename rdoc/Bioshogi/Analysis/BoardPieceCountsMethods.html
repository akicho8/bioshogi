def safe_delete_on(place)
  super.tap do |soldier|
    if soldier

      soldiers_count_per_location.decrement(soldier.location.key)
    end
  end

def soldiers_count
  @soldiers_count ||= Location.inject({}) { |a, e| a.merge(e.key => CounterHash.new) }
end
