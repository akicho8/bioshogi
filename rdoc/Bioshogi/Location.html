    # File lib/bioshogi/location.rb, line 23
def lookup(value)
  v = super

  unless v
    if value.kind_of?(Integer)
      v = lookup(value.modulo(count))
    end
  end

  unless v
    v = find { |e| e.match_target_values_set.include?(value) }
  end

  v
end

    lookup(-1).key # => :white lookup(:white).key # => :white
