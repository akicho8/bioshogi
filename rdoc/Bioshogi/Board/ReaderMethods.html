def blank_places
  Enumerator.new do |y|
    Place.each do |place|
      unless surface[place]
        y << place
      end
    end
  end
end

    to_sfen ()
    Source 
    # File lib/bioshogi/board/reader_methods.rb, line 80
def to_sfen
  Dimension::Row.dimension_size.times.collect { |y|
    Dimension::Column.dimension_size.times.collect { |x|
      lookup([x, y])
    }.chunk(&:class).flat_map { |klass, e|
      if klass == NilClass
        e.count
      else
        e.collect(&:to_sfen)
      end
    }.join
  }.join("/")
end

def vertical_soldiers(x)
  Enumerator.new do |yielder|
    Dimension::Row.dimension_size.times do |y|
      if soldier = lookup([x, y])
        yielder << soldier
      end
    end
  end
end
