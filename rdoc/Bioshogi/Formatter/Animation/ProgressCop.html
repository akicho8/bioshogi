def initialize(total = 100, params = {}, &callback)
  @total = total

  @params = {
    throttle_interval: 1,  # trigger? は最低N秒間隔とする。指定したからといって callback の間隔が変わるわけではない
  }.merge(params)

  @callback = callback

  @step = 0
  @message = nil

  @timestamp = nil
  @last_triggered = nil
end

def next_jump(step_value, message = nil)
  @timestamp = Time.now
  @message = message
  @step += step_value
  if interval_passed?
    @trigger = true
    @last_triggered = @timestamp
  else
    @trigger = false
  end
  Bioshogi.logger.info { log }
  if @callback
    @callback.call(self)
  end
end

def to_s
  [
    @timestamp.strftime("%F %T"),
    "#{@step}/#{@total}",
    "%6.2f %%" % percent,
    "T%d" % (@trigger ? 1 : 0),
    "#{message}"
  ].join(" ").strip
end
